class Биоинформатик: def __init__(self, имя, рост, вес, интеллект, скиллы, степень_бодрости): self.имя = имя self.рост = рост self.вес = вес self.интеллект = интеллект self.скиллы = скиллы self.степень_бодрости = степень_бодрости def ругаться_на_плохие_данные(self): print("Эти данные ужасны! Кто их вообще собирал?") def считать_до_миллиона(self): if self.степень_бодрости > 50: print("1, 2, 3, ... 1000000") else: print("Я слишком устал, чтобы считать до миллиона") def пить_кофий(self): if self.степень_бодрости < 15: print("К сожалению, ваш биоинформатик умер") else: print("Бахну кофейку") def писать_код(self): print("def main():\n\tprint('Hello, world!')") def рисовать_случайные_картинки(self): import random for i in range(self.степень_бодрости): print("*" * random.randint(1, 10)) def анализировать_данные(self): print("Данные проанализированы. Результаты будут готовы через час") def __str__(self): return f"Биоинформатик {self.имя} имеет рост {self.рост} см, вес {self.вес} кг, интеллект {self.интеллект} баллов, скиллы {self.скиллы} и степень бодрости {self.степень_бодрости}" #ну и как пример, биоинформатик1 = Биоинформатик("Иван", 180, 75, 120, ["Python", "R", "Bioinformatics"], 80) биоинформатик2 = Биоинформатик("Мария", 170, 60, 110, ["Java", "C++", "Machine Learning"], 60) биоинформатик1.ругаться_на_плохие_данные() биоинформатик2.считать_до_миллиона() биоинформатик1.писать_код() биоинформатик2.рисовать_случайные_картинки() биоинформатик1.анализировать_данные() биоинформатик2.пить_кофий() print(биоинформатик1) print(биоинформатик2)
